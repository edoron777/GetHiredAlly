Add a button in the CV Optimizer results page that shows how the system

1. BACKEND ENDPOINT - Add to cv_routes.py:

   @router.post("/api/cv/analyze-structure")
   async def analyze_cv_structure(file: UploadFile):
       """Return detailed block structure analysis for debugging"""
       from common.detection.block_detector import detect_cv_blocks
       
       # Read file content
       content = await file.read()
       cv_text = extract_text_from_file(content, file.filename)
       
       # Get block structure
       block_structure = detect_cv_blocks(cv_text)
       
       # Format for display
       result = {
           "total_sections_found": len(block_structure.sections) if block_structure else 0,
           "sections": []
       }
       
       if block_structure:
           for section in block_structure.sections:
               result["sections"].append({
                   "type": section.block_type.name if hasattr(section, 'block_type') else str(section.type),
                   "start_line": section.start_line,
                   "end_line": section.end_line,
                   "content_preview": section.content[:200] if hasattr(section, 'content') else "",
                   "items_count": len(section.items) if hasattr(section, 'items') else 0
               })
       
       return result
