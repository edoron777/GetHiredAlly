STEP 1: Get changes data from API
The API returns changes_json with before/after for each change.

STEP 2: Create highlight function

function highlightChanges(text, changes) {
  let highlightedText = text;
  
  changes.forEach(change => {
    const afterText = change.after;
    // Wrap changed text with highlight span
    highlightedText = highlightedText.replace(
      afterText,
      `<mark class="bg-yellow-200 px-1 rounded">${afterText}</mark>`
    );
  });
  
  return highlightedText;
}
