"""
Consistency Handler

Detects inconsistencies (date formats, bullet styles, etc.)

detection_config format:
{
    "type": "consistency",
    "check_what": "date_format",
    "target_section": "all",
    "format_patterns": {
        "month_year_full": "\\b(January|February|...)\\s+\\d{4}\\b",
        "month_year_abbr": "\\b(Jan|Feb|...)\\s+\\d{4}\\b",
        "mm_yyyy": "\\b\\d{1,2}/\\d{4}\\b"
    },
    "issue_when": "multiple_formats"
}
"""

import re
import logging
from typing import List, Dict

from .base import BaseHandler, DetectedIssue
from backend.common.detection.section_extractor import CVStructure
from ..cache import DetectionRule

logger = logging.getLogger(__name__)


class ConsistencyHandler(BaseHandler):
    """
    Handler for consistency-based detection.
    """
    
    def detect(
        self, 
        cv_text: str, 
        cv_structure: CVStructure, 
        rule: DetectionRule
    ) -> List[DetectedIssue]:
        """
        Detect consistency issues.
        """
        issues = []
        config = rule.detection_config
        
        # Get configuration
        check_what = config.get('check_what', 'date_format')
        target_section = config.get('target_section', 'all')
        format_patterns = config.get('format_patterns', {})
        issue_when = config.get('issue_when', 'multiple_formats')
        
        if not format_patterns:
            return issues
        
        # Get target text
        text = self.get_target_text(cv_structure, target_section)
        if not text:
            return issues
        
        # Find which formats are used
        formats_found: Dict[str, List[str]] = {}
        
        for format_name, pattern_str in format_patterns.items():
            try:
                pattern = re.compile(pattern_str, re.IGNORECASE)
                matches = pattern.findall(text)
                if matches:
                    formats_found[format_name] = matches
            except re.error as e:
                logger.warning(f"Invalid pattern for {format_name}: {e}")
        
        # Check for inconsistency
        formats_used = list(formats_found.keys())
        
        if issue_when == 'multiple_formats' and len(formats_used) > 1:
            match_text = f"Multiple {check_what} formats: {', '.join(formats_used)}"
            
            issue = self.create_issue(
                rule=rule,
                match_text=match_text,
                location=target_section,
                details={
                    'check_what': check_what,
                    'formats_found': {k: v[:5] for k, v in formats_found.items()},
                    'format_count': len(formats_used),
                    'target_section': target_section
                }
            )
            issues.append(issue)
        
        return issues
