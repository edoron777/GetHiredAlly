/**
 * Find the position of matchText in content using word boundaries.
 * For single words like "I", ensures we find standalone "I" not "I" inside other words.
 */
function findWordBoundaryMatch(content: string, matchText: string): number {
  if (!matchText || !content) return -1;
  
  const trimmedMatch = matchText.trim();
  
  // For very short matches (1-2 chars), use word boundary regex
  if (trimmedMatch.length <= 2) {
    // Escape special regex characters
    const escaped = trimmedMatch.replace(/[.*+?^${}()|[\]\\]/g, '\\$&');
    const wordBoundaryRegex = new RegExp(`\\b${escaped}\\b`, 'i');
    const match = content.match(wordBoundaryRegex);
    if (match && match.index !== undefined) {
      return match.index;
    }
    return -1;
  }
  
  // For longer matches, regular indexOf is fine
  return content.indexOf(trimmedMatch);
}

/**
 * Find ALL positions of matchText for multiple markers.
 */
function findAllWordBoundaryMatches(content: string, matchText: string): number[] {
  if (!matchText || !content) return [];
  
  const positions: number[] = [];
  const trimmedMatch = matchText.trim();
  
  // Escape special regex characters
  const escaped = trimmedMatch.replace(/[.*+?^${}()|[\]\\]/g, '\\$&');
  
  // Use word boundaries for short text, global search for all occurrences
  const flags = trimmedMatch.length <= 2 ? 'gi' : 'g';
  const pattern = trimmedMatch.length <= 2 ? `\\b${escaped}\\b` : escaped;
  const regex = new RegExp(pattern, flags);
  
  let match;
  while ((match = regex.exec(content)) !== null) {
    positions.push(match.index);
  }
  
  return positions;
}
