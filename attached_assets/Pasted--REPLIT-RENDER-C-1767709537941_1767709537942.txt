Create a utility function to clean markdown:
```tsx
// utils/cleanMarkdown.ts
export function cleanMarkdown(text: string): string {
  return text
    // Remove bold markers
    .replace(/\*\*([^*]+)\*\*/g, '$1')
    // Remove italic markers
    .replace(/\*([^*]+)\*/g, '$1')
    // Remove underscores for bold/italic
    .replace(/__([^_]+)__/g, '$1')
    .replace(/_([^_]+)_/g, '$1')
    // Remove heading markers
    .replace(/^#+\s*/gm, '')
    // Remove bullet points (keep text)
    .replace(/^[-*]\s+/gm, 'â€¢ ')
    // Clean up extra whitespace
    .replace(/\n{3,}/g, '\n\n')
    .trim();
}

// In StructureOverlay.tsx:

// Where content is rendered:

  {cleanMarkdown(sectionContent)}
</div>
```
