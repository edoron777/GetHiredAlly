async def check_smart_questions_eligibility(user_id: str) -> dict:

    user = await supabase.table("users").select("smart_questions_free_used").eq("id", user_id).single().execute()
    
    free_used = user.data.get("smart_questions_free_used", False) if user.data else False

    return {
        "eligible": is_paid or not free_used,

    }

async def mark_free_trial_used(user_id: str):

    await supabase.table("users").update({"smart_questions_free_used": True}).eq("id", user_id).execute()
