```typescript
/**
 * Determine if an issue should have a marker in the document.
 * 
 * Some issues can have markers (specific text found in CV).
 * Some issues cannot have markers (general issues like "CV too long").
 */
const isValidMarker = (
  issue: Issue,
  cvContent: string
): { valid: boolean; matchText: string; position: number } => {
  
  // Get the text to mark
  const matchText = issue.current || issue.matchText || '';
  
  // Issues that should NOT have markers (general/structural issues)
  const noMarkerIssues = [
    'LENGTH_CV_TOO_LONG',
    'LENGTH_CV_TOO_SHORT',
    'LENGTH_SUMMARY_TOO_LONG',
    'LENGTH_EXPERIENCE_TOO_DETAILED',
    'LENGTH_EDUCATION_TOO_DETAILED',
    'CONTENT_SUMMARY_TOO_LONG',
    'CONTENT_JOB_DESCRIPTION_TOO_LONG',
    'CONTENT_EDUCATION_DESCRIPTION_TOO_SHORT',
    'CONTENT_JOB_DESCRIPTION_TOO_SHORT',
  ];
  
  if (noMarkerIssues.includes(issue.issue_code)) {
    return { valid: false, matchText: '', position: -1 };
  }
  
  // "Missing" issues - mark the section where item should be
  if (matchText.startsWith('Not found in:')) {
    // For missing items, we could mark the contact section header
    // Or just don't show marker for missing items
    return { valid: false, matchText: '', position: -1 };
  }
  
  // Empty matchText - no marker
  if (!matchText || matchText.trim() === '') {
    return { valid: false, matchText: '', position: -1 };
  }
  
  // For very short matches (1-2 chars like "I"), use word boundary search
  const position = findWordBoundaryMatch(cvContent, matchText);
  
  if (position === -1) {
    return { valid: false, matchText: '', position: -1 };
  }
  
  return { valid: true, matchText: matchText.trim(), position };
};
```
