The function strip_structure_markers() removes all [H1], [BOLD], [BULLET] markers

SOLUTION:
Create a new function that converts markers to HTML instead of removing them.

STEP 1: Create new file backend/common/utils/marker_converter.py

import re

def convert_markers_to_html(text: str) -> str:
    """Convert structure markers to HTML elements."""
    if not text:
        return ""
    
    lines = text.split('\n')
    result_lines = []
    in_bullet_list = False
    
    for line in lines:
        is_bullet = '[BULLET]' in line
        
        # Handle bullet list open/close
        if is_bullet and not in_bullet_list:
            result_lines.append('<ul class="cv-bullets">')
            in_bullet_list = True
        elif not is_bullet and in_bullet_list:
            result_lines.append('</ul>')
            in_bullet_list = False
        
        # Convert [H1] to heading
        if '[H1]' in line:
            line = re.sub(r'\[H1\](.*)', r'<h2 class="cv-section">\1</h2>', line)
        
        # Convert [H2] to subheading
        elif '[H2]' in line:
            line = re.sub(r'\[H2\](.*)', r'<h3 class="cv-subsection">\1</h3>', line)
        
        # Convert [BULLET] to list item
        elif '[BULLET]' in line:
            line = re.sub(r'\[BULLET\](.*)', r'<li>\1</li>', line)
        
        # Regular paragraph
        elif line.strip():
            line = f'<p>{line}</p>'
        
        # Convert [BOLD] to strong
        line = line.replace('[BOLD]', '<strong>').replace('[/BOLD]', '</strong>')
        
        result_lines.append(line)
    
    if in_bullet_list:
        result_lines.append('</ul>')
    
    return '\n'.join(result_lines)
