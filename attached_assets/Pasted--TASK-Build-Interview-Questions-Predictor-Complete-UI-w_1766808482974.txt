-- Create category descriptions table
CREATE TABLE IF NOT EXISTS question_categories (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    category_key TEXT UNIQUE NOT NULL,
    category_name TEXT NOT NULL,
    description TEXT NOT NULL,
    question_format TEXT,
    answer_method TEXT,
    display_order INTEGER,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW()
);

ALTER TABLE question_categories ENABLE ROW LEVEL SECURITY;
CREATE POLICY "Anyone can read categories" ON question_categories FOR SELECT USING (true);
CREATE POLICY "Service can insert categories" ON question_categories FOR INSERT WITH CHECK (true);
CREATE POLICY "Service can delete categories" ON question_categories FOR DELETE USING (true);
