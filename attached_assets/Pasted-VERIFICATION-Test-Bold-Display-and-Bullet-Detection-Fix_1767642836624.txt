# Save and extract

doc.save(doc_bytes)
doc_bytes.seek(0)

result = _extract_docx_with_markers(doc_bytes.read())

print("=== EXTRACTION RESULT ===")
for line in result.split('\n'):
    if line.strip():
        print(line)
print("=== END ===")

# Count bullets
bullet_count = result.count('[BULLET]')
print(f"\n[BULLET] markers found: {bullet_count}")
print(f"Expected: 4 (or more)")

if bullet_count >= 4:
    print("✅ Bullet detection working!")
else:
    print("❌ Bullet detection needs work")
